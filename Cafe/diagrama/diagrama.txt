Table usuario {
  id int [pk, increment]
  correo varchar(100)
  contrasenia varchar(255)
  rol varchar(20)
}

Table token {
  id int [pk, increment]
  usuario_id int [ref: > usuario.id]
  token varchar(255)
  expiracion datetime
}

Table producto {
  id int [pk, increment]
  nombre varchar(100)
  precio int
  id_categoria int [ref: > categoria.id]
}

Table categoria {
  id int [pk, increment]
  nombre varchar(50)
}

Table carrito {
  id int [pk, increment]
  producto_id int [ref: > producto.id]
  cantidad int
}

Table pedido {
  id int [pk, increment]
  nombre varchar(100)
  correo varchar(100)
  id_metodo int [ref: > metodo.id]
  productos text
  total int
  fecha datetime
}

Table metodo {
  id int [pk, increment]
  nombre char(8)
}
